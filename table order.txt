CREATE TABLE ORDER_ITEM
(
	ID NUMERIC(3) CONSTRAINT ORDER_ITEM_PK PRIMARY KEY,
  	 ORDERID NUMERIC(3) CONSTRAINT ORDERS_FK REFERENCES ORDERS(ID),  
       	 PRODUCTID NUMERIC(3) CONSTRAINT PRODUCT_FK REFERENCES PRODUCT(ID),  
         UNITPRICE   NUMERIC(7,2), 
                  QUANTITY   NUMERIC(3)


);
